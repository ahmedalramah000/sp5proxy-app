# SP5Proxy VPS Services Fix Commands
# ==================================
# SSH to VPS: ssh root@168.231.82.24
# Then run these commands:

# 1. Check current status
pm2 list
netstat -tlnp | grep -E ':(8080|3002|3000)'

# 2. Stop and clean all services
pm2 stop all
pm2 delete all

# 3. Kill processes on ports (if needed)
fuser -k 8080/tcp
fuser -k 3002/tcp
fuser -k 3000/tcp

# 4. Navigate to project
cd /root/sp5proxy

# 5. Start Admin Panel
cd admin-panel
pm2 start server.js --name sp5proxy-admin -- --port 8080
cd ..

# 6. Start API Server
pm2 start src/api-server.js --name sp5proxy-api

# 7. Start Web Server
pm2 start server-web.js --name sp5proxy-web

# 8. Save PM2 configuration
pm2 save
pm2 startup

# 9. Check final status
pm2 list
netstat -tlnp | grep -E ':(8080|3002|3000)'

# 10. View logs
pm2 logs --lines 20
